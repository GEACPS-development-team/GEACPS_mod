JAP_parliament_reload_effect = {
	clear_array = JAP_parliament_array

	for_each_loop = {
		array = JAP_parliament_seat_array
		add_to_array = { JAP_seat = JAP_parliament_seat_array^i }
	}

	set_temp_variable = { temp_first_index = 0 }
	set_temp_variable = { temp_second_index = 1 }

	set_temp_variable = { temp_size = JAP_seat^num }
	subtract_from_temp_variable = { temp_size = 1 }

	while_loop_effect = {
		limit = { check_variable = { temp_first_index < temp_size } }
		add_to_variable = { JAP_seat^temp_second_index = JAP_seat^temp_first_index }
		add_to_temp_variable = { temp_first_index = 1 }
		add_to_temp_variable = { temp_second_index = 1 }
	}

	for_loop_effect = {
		start = 0
		end = JAP_seat^num

		resize_array = {
			array = JAP_parliament_array
			size = JAP_seat^v
			value = JAP_parliament_group_array^v
		}
	}

	clear_array = JAP_seat
}
