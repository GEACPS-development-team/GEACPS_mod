se_back_map_setting = {
	if = {
		limit = {
			any_state = {
				is_owned_by = ROOT
				is_capital = yes
				OR = {
					is_on_continent = asia
					is_on_continent = australia
				}
			}
		}
		set_variable = {	special_event_back_map = 1	}
	}
	else_if = {
		limit = {
			any_state = {
				is_owned_by = ROOT
				is_capital = yes
				OR = {
					is_on_continent = europe
					is_on_continent = africa
				}
			}
		}
		set_variable = {	special_event_back_map = 2	}
	}
	else_if = {
		limit = {
			any_state = {
				is_owned_by = ROOT
				is_capital = yes
				is_on_continent = north_america
			}
		}
		set_variable = {	special_event_back_map = 3	}
	}
	else_if = {
		limit = {
			any_state = {
				is_owned_by = ROOT
				is_capital = yes
				is_on_continent = south_america
			}
		}
		set_variable = {	special_event_back_map = 4	}
	}
	else = {
		set_variable = {	special_event_back_map = 0	}
	}
}

se_ZZZ_test_event = {
	every_country = {
		limit = {	is_ai = no	}
		play_song = "Test_Song_1"
		set_variable = {	special_event_ID = 1	}
		set_variable = { special_event_clock_time_hour = 15 }
		se_back_map_setting = yes
	}
}
