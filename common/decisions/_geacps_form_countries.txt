formation_decisions = {
	form_Africa_Union = {
		visible = {
			always = no
			any_state = {
				state = ROOT.capital
				is_on_continent = africa
			}
		}
		complete_effect = {
			set_cosmetic_tag = AFRICA_union
			every_state = {
				limit = {	is_on_continent = Africa	}
				add_core_of = ROOT
			}
		}
	}
	form_EUROPE_union = {
		visible = {
			NOT = {	original_tag = EUR	}
			ROOT = {
				capital_scope = {	is_on_continent = europe	}
				all_of_scopes = {
					array = Global.eu_founder_state_array
					is_owned_by = ROOT
				}
			}
		}
		available = {
			NOT = {	original_tag = EUR	}
			ROOT = {
				capital_scope = {	is_on_continent = europe	}
				all_of_scopes = {
					array = Global.eu_founder_state_array
					is_owned_by = ROOT
				}
			}
		}
		complete_effect = {
			every_state = {
				limit = {	is_on_continent = europe	}
				add_core_of = EUR
			}
			every_country = {
				limit = {	is_subject_of = ROOT	}
				if = {
					limit = {	capital_scope = {	is_on_continent = europe	}	}
					EUR = {	annex_country = {	target = PREV	transfer_troops = yes	}	}
				}
				else = {	EUR = {	puppet = PREV	}	}
			}
			EUR = {
				annex_country = {	target = ROOT	transfer_troops = yes	}
				change_tag_from = ROOT
				set_cosmetic_tag = EUR_union
			}
		}
	}
}
