	on_startup = {#中華民国用ゲームスタート
		effect = {
			random_country = {
				limit = {
					is_ai = no

				}
				country_event = { id = GEACPS_game_start.1 }

		}

	on_state_control_changed = {#武漢陥落後のイベントとコスメティックタグ設定

			if = {
				limit = {
					FROM = {
						tag = CHI
					}
					ROOT = {
						is_in_faction_with = JAP
					}
					FROM.FROM = { 
							state = 620
					}
					NOT = { has_global_flag = fall_of_Wuhan }
				}
				FROM = {
					country_event = {
						id = GEACPS_china.003
						days = 1
					}
					set_cosmetic_tag = CHI_republic_chongqing_goverment
				}
				set_global_flag = fall_of_Wuhan

			}
		}
	}

			if = {

						tag = SCH
