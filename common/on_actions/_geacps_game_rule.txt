				limit = {

						check_variable = {	FRA_state_variable = 1	}

						check_variable = {	ITA_state_variable = 1	}
						check_variable = {	NTL_state_variable = 1	}
						check_variable = {	BEL_state_variable = 1	}
						check_variable = {	LUX_state_variable = 1	}
					}
				}
				add_to_array = {
					array = Global.eu_founder_state_array
					value = THIS
				}
			}
		}
	}
	on_startup = {
		effect = {
			if = {
				limit = {
					has_game_rule = {
						rule = annex_colonial_states
						option = OPTION_ANNEX_COLONIAL_STATES
					}
				}
				every_country = {
					limit = {
						OR = {
							has_autonomy_state = overseas_territory

							has_autonomy_state = SOV_constituent_republic
							has_autonomy_state = SOV_constituent_federal_republic
						}
					}
					OVERLORD = {
						annex_country ={
							target = PREV
							transfer_troops = yes
						}
					}
					every_owned_state = {
						OVERLORD = {	add_state_claim = PREV	}
					}
				}
				SUD = {
					every_owned_state = {
						EGY = {
							transfer_state = PREV
							add_state_claim = PREV
						}
					}
				}
			}
		}
	}
